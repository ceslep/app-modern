    e.preventDefault(); if (!_svl_animate) return;
    divContents.forEach((div) => div.classList.add("d-none"));
    resetAllTables();
    document
        .getElementById("concentradorInasistencias")
        .classList.remove("d-none");
    document.getElementById("navbarNavAltMarkup").classList.remove("show");
    document.getElementById("controli").classList.remove("show");
    document.getElementById("navbarNavAltMarkup").click();
});



document
    .querySelector("#frmConcentradorInasistencias")
    .addEventListener("submit", async (e) => {
        e.preventDefault(); if (!_svl_animate) return;
        document.querySelector("#mostrarRetardos").checked = false;
        document.querySelector("#mostrarExcusas").checked = false;
        document.querySelector(".bi-card-list").classList.add("d-none");
        document.querySelector(".spnconcinas").classList.remove("d-none");
        let data = new FormData(e.target);
        data = serialize(data);
        PERIODO = data.periodo;
        consolelog(data);
        document.querySelector(".concentradorInasistencias").innerHTML = spnG;
        let html = await getConcentrador(data, "inasistencia");
        document.querySelector(".concentradorInasistencias").innerHTML = html;
        let row0 = document.getElementById("tableconcentradorinasistencia").rows[0];

        for (let i = 1; i < row0.cells.length; i++) {
            if (row0.cells[i].children[0]) {
                let textContent = row0.cells[i].children[0].textContent.trim();
                let asignatura = JSON.stringify({ ...data, asignatura: textContent });
                row0.cells[i].children[0].innerHTML = `
               
            <button class="btn btn-sm btn-inas-est" data-asignatura='${asignatura}'>
                <i class="cntgraphic2t">${textContent}</i>
                <div class="spinner-border text-center spngraphic2t d-none"
                            role="status" style="width: 1.3rem; height: 1.3rem;">
                            <span class="visually-hidden">Loading...</span>
                        </div>
            </button>
                        
        `;
            }
        }
        document.querySelector(".bi-card-list").classList.remove("d-none");
        document.querySelector(".spnconcinas").classList.add("d-none");

        let response = await fetch("getInasistencias.php", {
            method: "POST",
            body: JSON.stringify(data),
            headers: {
                "Content-Type": "application/json",
            },
        });
        let inasistencias = await response.json();
        if (inasistencias.length > 0) {
            document.getElementById("mostrarRetardos").removeAttribute("disabled");
            document.getElementById("mostrarExcusas").removeAttribute("disabled");
        }
        if (!inasistencias.length)
            document.getElementById("mostrarExcusas").removeAttribute("disabled");
        await inasistencias.forEach(async (inasistencia) => {
            let { estudiante, asignatura, cantidadInasistencias } = inasistencia;
            try {
                var idest = `estudiante_${estudiante}_${asignatura}`;
                document.getElementById(idest).innerHTML = `
        <a href="#!" data-inasistencia='${JSON.stringify({
                    ...inasistencia,
                    periodo: PERIODO,
                })}' data-retardos='N' data-excusas='N'>
        <span class="fs-3 cnt">${cantidadInasistencias}</span>
        <div class="spinner-border spinner-border-sm spninasa d-none" role="status">
                                <span class="visually-hidden"></span>
                            </div>
        </a>

        `;
            } catch (e) {
                consolelog(`${e} en id=${idest}`);
            }
        });
    });
